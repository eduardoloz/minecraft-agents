JavaScript Helper Function

function renderHearts(container, health, maxHealth = 20) {
    container.innerHTML = '';
    const totalHearts = maxHealth / 2; // 20 health = 10 hearts
    const fullHearts = Math.floor(health / 2);
    const halfHeart = health % 2 === 1;
    
    // Full hearts
    for (let i = 0; i < fullHearts; i++) {
        const heart = document.createElement('div');
        heart.className = 'heart full';
        container.appendChild(heart);
    }
    
    // Half heart
    if (halfHeart) {
        const heart = document.createElement('div');
        heart.className = 'heart half';
        container.appendChild(heart);
    }
    
    // Empty hearts
    const emptyHearts = totalHearts - fullHearts - (halfHeart ? 1 : 0);
    for (let i = 0; i < emptyHearts; i++) {
        const heart = document.createElement('div');
        heart.className = 'heart empty';
        container.appendChild(heart);
    }
}
